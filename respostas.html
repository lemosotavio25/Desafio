Desafio 1: Sequência de Fibonacci


function isFibonacci(num) {
    if (num < 0) return false;

    let a = 0;
    let b = 1;

    if (num === a || num === b) return true;

    let c = a + b;

    while (c <= num) {
        if (c === num) return true;

        a = b;
        b = c;
        c = a + b;
    }

    return false;
}

const numToCheck = 21;

if (isFibonacci(numToCheck)) {
    console.log(`O número ${numToCheck} pertence à sequência de Fibonacci.`);
} else {
    console.log(`O número ${numToCheck} NÃO pertence à sequência de Fibonacci.`);
}

Desafio 2: Verificação de Letras na String


function countAInString(str) {
    let count = 0;
    let lowerCaseStr = str.toLowerCase();

    for (let i = 0; i < lowerCaseStr.length; i++) {
        if (lowerCaseStr[i] === 'a') {
            count++;
        }
    }

    if (count > 0) {
        console.log(`A letra 'a' aparece ${count} vezes na string.`);
    } else {
        console.log("A letra 'a' não foi encontrada na string.");
    }
}

const stringToCheck = "JavaScript é uma linguagem fantástica!";
countAInString(stringToCheck);
